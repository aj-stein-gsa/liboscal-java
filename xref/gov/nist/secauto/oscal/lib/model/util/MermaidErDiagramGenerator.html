21  import java.io.IOException;
22  import java.io.PrintWriter;
23  
24  import edu.umd.cs.findbugs.annotations.NonNull;
25  
26  public class MermaidErDiagramGenerator
27      extends AbstractRecursionPreventingNodeItemVisitor<PrintWriter, Void> {
28  
29    private enum Relationship {
30      ZERO_OR_ONE("|o", "o|"),
31      ONE("||", "||"),
32      ZERO_OR_MORE("}o", "o{"),
33      ONE_OR_MORE("}|", "|{");
34  
35      @NonNull
36      private final String left;
37      @NonNull
38      private final String right;
39  
40      Relationship(@NonNull String left, @NonNull String right) {
41        this.left = left;
42        this.right = right;
43      }
44  
45      @NonNull
46      public String getLeft() {
47        return left;
48      }
49  
50      @NonNull
51      public String getRight() {
52        return right;
53      }
54  
55      @NonNull
56      public static Relationship toRelationship(int minOccurs, int maxOccurs) {
57        return minOccurs < 1
58            ? maxOccurs == 1
59                ? ZERO_OR_ONE
60                : ZERO_OR_MORE
61            : maxOccurs == 1
62                ? ONE
63                : ONE_OR_MORE;
64      }
65  
66      private String generate() {
67        return getLeft() + "--" + getRight();
68      }
69    }
70  
71    @Override
72    public Void visitField(IFieldNodeItem item, PrintWriter writer) {
73      processNodeItem(item, writer);
74      return super.visitField(item, writer);
75    }
76  
77    @Override
78    public Void visitAssembly(IAssemblyNodeItem item, PrintWriter writer) {
79      processNodeItem(item, writer);
80      return super.visitAssembly(item, writer);
81    }
82  
83    @Override
84    public Void visitAssembly(IAssemblyInstanceGroupedNodeItem item, PrintWriter writer) {
85      processNodeItem(item, writer);
86      return super.visitAssembly(item, writer);
87    }
88  
89    @Override
90    protected Void defaultResult() {
91      return null;
92    }
93  
94    private void processNodeItem(@NonNull IModelNodeItem<?, ?> item, PrintWriter writer) {
95      IAssemblyNodeItem left = item.getParentContentNodeItem();
96  
97      IAssemblyDefinition leftDefinition = left.getDefinition();
98      INamedModelInstance rightInstance = item.getInstance();
99      IModelDefinition rightDefinition = item.getDefinition();
100 
101     Relationship relationship
102         = Relationship.toRelationship(rightInstance.getMinOccurs(), rightInstance.getMaxOccurs());
103 
104     writer.format("  %s %s %s : %s%n",
105         leftDefinition.getEffectiveName(),
106         relationship.generate(),
107         rightDefinition.getEffectiveName(),
108         rightInstance.getEffectiveName());
109   }
110 
111   public void generate(@NonNull IBoundModule module, @NonNull PrintWriter writer) throws IOException {
112     IModuleNodeItem moduleItem = INodeItemFactory.instance().newModuleNodeItem(module);
113 
114     for (IAssemblyDefinition root : module.getExportedRootAssemblyDefinitions()) {
115       for (IModelNodeItem<?, ?> rootItem : moduleItem.getModelItemsByName(root.getDefinitionQName())) {
116         IAssemblyDefinition definition = (IAssemblyDefinition) rootItem.getDefinition();
117 
118         writer.println("erDiagram");
119         writer.format("  module ||--|| %s: \"%s\"%n", definition.getEffectiveName(), definition.getRootName());
120 
121         for (IModelNodeItem<?, ?> modelItem : CollectionUtil.toIterable(rootItem.modelItems())) {
122           modelItem.accept(this, writer);
123         }
124       }
125     }
126   }
127 }
